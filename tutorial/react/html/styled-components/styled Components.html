App.js 
import { Container } from ".components/styled/Container.styled"

//Note that we put it in curly braces. This is because we didn't say export default for it. So it is not the default export 


function App() {
    return (
        <Container>
            <h1>Hello</h1>
        </Container>
    )
}

export default App;

//You'll see that when it is compiled, it will be compiled to a div

      import styled from "styled-components"

          max-width: 100%;
          padding: 0 20px;
          margin: 0 auto;

          h1 {
              color: red;
          }
          &:hover {
              background-color: green;
          }
      `
