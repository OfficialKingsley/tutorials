useReducer hook

This is actually a replacement for the useState() hook. Both hooks actually have the same purpose.
However, in a situation where you have a trigger action (maybe a click event on a button) and this action is supposed to perform two or more events (i.e. change two or more states), you might consider the use reducer hook so that you don't have to create multiple states 

import {useState} from react;

const ReducerTutorial= () => {
    const [count, setCount] = useState(0); //Notice that the useState() hook was used in the component
    const [showText, setShowText] = useState(true);

    return (
        <div>
        <h1>{count}</h1>
            <button 
            onClick={()=> {
                setCount(count + 1)
                setShowText(!showText)
            }}
            >
                Increment
            </button>

            {showText && <p>This is a text</p>}
        </div>
    )

}

                
import {useReducer} from react;

const reducer = (state, action) => {
    switch (action.type) {  //Notice the use of the switch statement
        case "INCREMENT":  //INCREMENT is  the name of an action
            return {count: state.count + 1, showText: !state.showText}  //Notice that we also returned all the states
        case "toggleshowtext":
            return {count: state.count, showText: !state.showText}
        default: //This is used if we try to access a function that wasn't defined
            return state
    }
}
//The above is a reducer function that:
//Uses the switch conditional statement
const ReducerTutorial= () => {
    const [state, dispatch] = useReducer(reducer,  {count: 0, showText: true})
    
    
    return (
        <div>
        <h1>{state.count}</h1>
            <button 
            onClick={()=> {
                dispatch({type: "INCREMENT"});  //The dispatch  function accepts the action parameter
                dispatch({type: "toggleshowtext"})
                //See how the type variable of the action attribute was passed into the dispatch function
            }}
            >
                Increment
            </button>

            {state.showText && <p>This is a text</p>}
        </div>
    )

}
