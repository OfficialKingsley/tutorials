Using the react-router-dom

To use react-router, we have to install react-router-dom. When it has been installed, we have to wrap everything in the highest component which is the App.js with the <BrowserRouter></BrowserRouter> tag
We do this so that we can create routes in the App.js

App.js 
import React from "react";
import { BrowserRouter, Route, Switch } from "react-router-dom";
import About from "./components/About";
import Blog from "./components/Blog";
import Contact from "./components/Contact";
import Home from "./components/Home";

function App() {
    return (
        <BrowserRouter>
            <div className="App">
                <Home />
                <About />
                <Contact />
                <Blog />
            </div>
        </BrowserRouter>
    );
}

export default App;

To make a component a route, we have to wrap it around the <Route></Route> tag as such 


import React from "react";
import { BrowserRouter, Route, Switch } from "react-router-dom";
import About from "./components/About";
import Blog from "./components/Blog";
import Contact from "./components/Contact";
import Home from "./components/Home";

function App() {
    return (
        <<BrowserRouter>
            <<div className="App">
                <<Route path="/home">
                    <<Home />
                <</Route>
                <<Route path="/about">
                    <<About />
                <</Route>

                <<Route path="/contact">
                    <<Contact />
                <</Route>
                <<Route path="/blog">
                    <<Blog />
                <</Route>
            <</div>
        </BrowserRouter>
    );
}

export default App;

The path attribute signifies the route in the url
