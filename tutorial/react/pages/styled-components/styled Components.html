Using themes in styled components

To do this, we import the ThemeProvider from styled components

App.js 
        import { Container } from ".components/styled/Container.styled"
        import { ThemeProvider } from "styled-components";
        //Notice that it was imoprted in the App.js
        //The ThemeProivider takes in a prop called theme which accepts an object
        
        const theme = {
            colors: {
                header: "#ebfbff"
                body: "#fff"
                footer: "#003333"
            }
        }
        function App() {
            return (
                <ThemeProvider theme={theme}>
                    <Container bg="red">
                        <h1>Hello</h1>
                    </Container>
                </ThemeProvider>
            )
            //So we wrap everything around the ThemeProvider
        }
        
        export default App;
        
        //You'll see that when it is compiled, it will be compiled to a div
