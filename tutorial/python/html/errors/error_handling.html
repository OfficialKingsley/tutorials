The try except helps us to handle errors before the code even runs so that our code doesn't break 

x = int(input("Input an integer"))
print(x)

try:
    x = int(input("Input an integer"))
    print(x)
except:
    print("Something went wrong... Please try again")
                
try:
    x = int(input("Input an integer"))
    print(x)
except ValueError:
    print("Something went wrong... Please try again")
                
try:
    x = int(input("Input an integer"))
    print(x)
except ValueError:
    print("Something went wrong... Please try again")
else:
    print("Nothing went wrong")
    #If everything is okay
                
try:
    x = int(input("Input an integer"))
    print(x)
except ValueError:
    print("Something went wrong... Please try again")
finally:
    print("Try except finished")
    #This will run regardless of if there's an error or not
