hobbies = ["Sports", "Cooking"]

//What typescript will infer is that 
//hobbies is an array of strings which will be written as

hobbies: string[]
      
let favoriteActivities: string[];

favoriteActivities = "Sports" //Gives error
favoriteActivities = ["sports", 1]; //Gives error
favoriteActivities = ["sports"] //No error

If you want it to accept multiple data types, you can do the following 

let favoriteActivities: any[];

favoriteActivities = "Sports" //Gives error
favoriteActivities = ["sports", 1]; //No error
favoriteActivities = ["sports"] //No error

Another advantage of typescript inference is that when we loop through, it automatically detects what type of data each item in the iterable is 


const hobbies = ["Football","Coding"]
for (const hobby of hobbies) {
  console.log(hobby)
}
//Typescript automatically infers that the hobby is a string
