Creating HTTP servers with node js

const http = require("http");
const server = http.createServer((req, res)=>{
  req - request - this is what the client requests from the server
  After we get the response we can get send back a response
  res.write("Hello world from js"); This is the response we wanna send
  res.end() Now, we've sent the response
})

//To start the server
//server.listen(port-number)
server.listen("3000")

Serving Static Files with Http and File system modules

const http = require("http");
const fs = require("fs");
const server = http.createServer((req, res)=> {
  const readStream =fs.createReadStream("./static/index.html")
  res.writeHead(200, {
    "Content-type": "text/html"
  });
  readStream.pipe(res)

  //The res is also a writable stream
})
server.listen(3000)

//Content types
//html- text/html
//json - application/json
